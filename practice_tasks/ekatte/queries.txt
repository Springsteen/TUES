CREATE TABLE region(
name varchar(200),
region_name varchar(10) primary key,
existence boolean
);

CREATE TABLE area (
name varchar(200),
region varchar(10) references region(region_name),
area_abbreviation varchar(10) primary key,
existence boolean
);

CREATE TABLE municipality (
name varchar(200),
mun_abbreviation varchar(10) primary key,
existence boolean
);

CREATE TABLE town_hall (
name varchar(200),
ekatte varchar(30),
town_hall_abbreviation varchar(20) primary key,
existence boolean
);

CREATE TABLE tsb (
name varchar(200),
tsb_number varchar(10) primary key,
existence boolean
);

CREATE TABLE document (
name varchar(500),
doc_number varchar(10) primary key,
doc_kind varchar(100),
doc_inst varchar(300),
existence boolean
);

CREATE TABLE location (
name varchar(200),
area varchar(10) references area(area_abbreviation),
municipality varchar(10) references municipality(mun_abbreviation),
existence boolean,
ekatte varchar(30) primary key,
town_hall_abbreviation varchar(20) references town_hall(town_hall_abbreviation), 
tsb_number varchar(10) references tsb(tsb_number),
doc_number varchar(20) references document(doc_number)
);

CREATE TABLE district (
district_num varchar(20) primary key,
name varchar(100),
doc_number varchar(20) references document(doc_number),
existence boolean
);

CREATE TABLE sofia_district (
ekatte varchar(30),
name varchar(100),
district_num varchar(20) references district(district_num),
existence boolean
);

CREATE TABLE resort (
ekatte varchar(30),
name varchar(200),
area varchar(200),
doc_number varchar(20) references document(doc_number),
existence boolean 
);
